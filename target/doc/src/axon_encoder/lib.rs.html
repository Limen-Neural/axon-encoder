32/// The core trait for all encoders in this crate.
33///
34/// Encoders convert continuous analog values into discrete spike events for
35/// spiking neural networks (SNNs). Two modes are supported:
36///
37/// - **Batch mode** (`encode`): Process a complete input vector at once
38/// - **Streaming mode** (`encode_step`): Process incrementally, one step at a time
39///
40/// # Example
41///
42/// ```rust
43/// use axon_encoder::prelude::*;
44///
45/// let mut encoder = RateEncoder::new(5.0, 50.0, (0.0, 1.0));
46/// let input = [0.25, 0.75, 0.5];
47///
48/// // Batch encoding
49/// let output = encoder.encode(&input);
50///
51/// // Reset for streaming (if using stateful encoder)
52/// encoder.reset();
53/// ```
54pub trait Encoder {
55    /// Encodes a slice of analog values into spike events (batch mode).
56    fn encode(&mut self, input: &[f32]) -> EncodedOutput;
57
58    /// Encodes a single step incrementally (streaming mode).
59    ///
60    /// By default, this delegates to `encode()` for stateless encoders.
61    /// Stateful encoders should override this to maintain state between calls.
62    ///
63    /// # Arguments
64    ///
65    /// * `input` - A slice of analog values to encode
66    ///
67    /// # Returns
68    ///
69    /// An `EncodedOutput` containing any spike events generated in this step
70    fn encode_step(&mut self, input: &[f32]) -> EncodedOutput {
71        self.encode(input)
72    }
73
74    /// Resets the encoder to its initial state.
75    fn reset(&mut self);
76}
