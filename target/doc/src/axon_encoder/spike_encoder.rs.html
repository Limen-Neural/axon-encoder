5    thresholds: Vec<f32>,

8impl DerivativeEncoder {
9    /// Creates a new derivative encoder with specific thresholds for each channel
10    pub fn new(thresholds: Vec<f32>) -> Self {
11        Self {
12            last_values: vec![0.0; thresholds.len()],
13            thresholds,
14        }
15    }
16
17    pub fn encode_step(&mut self, current_values: &[f32]) -> EncodedOutput {
18        let mut output = EncodedOutput::new();
19
20        for (i, &current_val) in current_values.iter().enumerate() {
21            if i >= self.thresholds.len() {
22                break;
23            }
24
25            let delta = current_val - self.last_values[i];
26
27            // Excitatory spike on positive jump exceeding threshold
28            if delta > self.thresholds[i] {
29                output.spikes.push(SpikeEvent {
30                    channel: i as u16,
31                    timestamp: 0,
32                    polarity: true,
33                });
34            }
35            // Inhibitory/Negative spike on sudden drop
36            else if delta < -self.thresholds[i] {
37                output.spikes.push(SpikeEvent {
38                    channel: i as u16,
39                    timestamp: 0,
40                    polarity: false,
41                });
42            }
43
44            self.last_values[i] = current_val;
45        }
46        output
47    }
